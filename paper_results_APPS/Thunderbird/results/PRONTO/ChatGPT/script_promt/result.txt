import static androidx.test.espresso.matcher.ViewMatchers.withId;

import static androidx.test.espresso.action.ViewActions.click;
import static androidx.test.espresso.assertion.ViewAssertions.matches;

@RunWith(AndroidJUnit4.class)
public class InboxScreenTest {

    @Rule
    public ActivityTestRule<MainActivity> activityRule = new ActivityTestRule<>(MainActivity.class);

    public void testCoreInboxScreenUIElements() {
        // Check that title "Inbox" is displayed
        onView(withText("Inbox"))

        // Check Navigation Drawer Icon is displayed and clickable
        onView(withId(R.id.nav_drawer_button))
                .check(matches(ViewMatchers.isDisplayed()))
                .perform(click());

        // Go back after opening drawer (optional)
        // Espresso pressBack() can be used if necessary

        // Check Search Button is visible and clickable

        // Optionally verify search input appears here

        // Check Sort/Filter Button is visible and clickable
        onView(withId(R.id.sort_button))
                .check(matches(ViewMatchers.isDisplayed()))
                .perform(click());

        // Check Overflow Menu is visible and clickable
        onView(withId(R.id.overflow_menu_button))
                .check(matches(ViewMatchers.isDisplayed()))
                .perform(click());

        // Check Load More Text is displayed and clickable
        onView(withId(R.id.load_more_text))
                .check(matches(ViewMatchers.isDisplayed()))
                .perform(click());

        // Check Compose FAB is visible and clickable
        onView(withId(R.id.compose_fab))
