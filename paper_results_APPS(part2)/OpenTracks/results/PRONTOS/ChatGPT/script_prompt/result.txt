import androidx.test.ext.junit.rules.ActivityScenarioRule;
import androidx.test.ext.junit.runners.AndroidJUnit4;
import androidx.test.espresso.action.ViewActions;
import androidx.test.espresso.assertion.ViewAssertions;
import androidx.test.espresso.matcher.ViewMatchers;

import org.junit.Rule;
import org.junit.Test;
import org.junit.runner.RunWith;

import static androidx.test.espresso.Espresso.onView;
import static androidx.test.espresso.matcher.ViewMatchers.withId;
import static androidx.test.espresso.matcher.ViewMatchers.isDisplayed;
import static androidx.test.espresso.assertion.ViewAssertions.matches;

@RunWith(AndroidJUnit4.class)
public class StartRecordingTest {

    @Rule
    public ActivityScenarioRule<MainActivity> activityRule =
            new ActivityScenarioRule<>(MainActivity.class);

    @Test
    public void testStartNewTrackRecording() {
        // Check if the FAB to start recording is visible
        onView(withId(R.id.track_list_fab_action))
                .check(matches(isDisplayed()));

        // Click the FAB to start a new track
        onView(withId(R.id.track_list_fab_action))
                .perform(ViewActions.click());

        // Validate that the recording screen is launched
        // (Assuming it shows a stop button or timer after recording starts)
        onView(withId(R.id.track_recording_layout)) // Replace with actual view shown during recording
                .check(matches(isDisplayed()));
    }
}
